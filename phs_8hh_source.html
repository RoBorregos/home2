19 template <std::size_t dimension>
20 inline auto phs_measure(float d_foci, float d_transverse) noexcept -> float
21 {
22 const float conjugate_diameter = std::sqrt(d_transverse * d_transverse - d_foci * d_foci);
23
24 float lmeas = d_transverse / 2.0;
25 for (auto i = 1U; i < dimension; ++i)
26 {
27 lmeas = lmeas * conjugate_diameter / 2.0;
28 }
29
30 return lmeas * unit_ball_measure(dimension);
31 }

108 inline void update_rotation() noexcept
109 {
110 // If the transverse diameter is too close to 0, we treat this as a circle.
111 static constexpr float circle_tolerance = 1e-6;
112 if (min_transverse_diameter < circle_tolerance)
113 {
114 rot_world_from_ellipse.setIdentity(dimension, dimension);
115 }
116 else
117 {
118 const EigenVector f1 = vamp::vector_to_eigen(focus_1);
119 const EigenVector f2 = vamp::vector_to_eigen(focus_2);
120 const EigenVector transverse_axis = (f2 - f1) / min_transverse_diameter;
121 const EigenMatrix wahba_prob = transverse_axis * EigenMatrix::Identity().col(0).transpose();
122
123 Eigen::JacobiSVD<EigenMatrix, Eigen::NoQRPreconditioner> svd(
124 wahba_prob, Eigen::ComputeFullV | Eigen::ComputeFullU);
125
126 EigenVector middle_m = EigenVector::Ones();
127 middle_m(dimension - 1) = svd.matrixU().determinant() * svd.matrixV().determinant();
128 rot_world_from_ellipse = svd.matrixU() * middle_m.asDiagonal() * svd.matrixV().transpose();
129 }
130 }
131
132 void update_transformation()
133
134 {
135 const float conjugate_diamater = std::sqrt(
136 transverse_diameter * transverse_diameter -
137 min_transverse_diameter * min_transverse_diameter);
138
139 EigenVector diag = EigenVector::Constant(0.5 * conjugate_diamater);
140 diag(0) = 0.5 * transverse_diameter;
141
142 tf_world_from_ellipse = rot_world_from_ellipse * diag.asDiagonal();
143 phs_measure = utils::phs_measure<dimension>(min_transverse_diameter, transverse_diameter);
144 }
