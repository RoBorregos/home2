56 // Get r-disc neighbours, then iterate through list until a valid connection is found
57 // Necessary workaround given asymmetric cost function
58 //* ------------------ ------ -------------------
59 // Only need to check nodes that are closer than the root of the tree, since connecting to the
60 // root will always be valid
61 inline auto find_nearest(NN *nn, const NNNode &root, const Configuration &c, float cost)
62 -> std::pair<NNNode, float>
63 {
64 std::vector<NNNode> near_list;
65
66 // Almost always just pulls in the entire graph, but good to be principled.
67 near_list.reserve(nn->size());
68
69 auto temp_node = NNNode{0, cost, c};
70 nn->nearestR(temp_node, NNNode::distance(temp_node, root), near_list);
71
72 const auto *new_nearest_node = &near_list[0];
73 float new_nearest_distance = c.distance(new_nearest_node->array);
74
75 for (auto idx = 1U; new_nearest_node->cost > 0 //
76 and cost < new_nearest_node->cost + new_nearest_distance //
77 and idx < near_list.size();
78 ++idx)
79 {
80 new_nearest_node = &near_list[idx];
81 new_nearest_distance = c.distance(new_nearest_node->array);
82 }
83
84 return {*new_nearest_node, new_nearest_distance};
85 }

99 inline auto solve(
100 const Configuration &start,
101 const std::vector<Configuration> &goals,
102 const collision::Environment<FloatVector<rake>> &environment,
103 const AORRTCSettings &settings,
104 const float max_cost,
105 typename RNG::Ptr rng) noexcept -> PlanningResult<Robot>
106 {
107 static constexpr std::size_t start_index = 0;
108 const RRTCSettings &rrtc_settings = settings.rrtc;
109 PlanningResult<Robot> result;
110
111 NN start_tree;
112 NN goal_tree;
113
114 std::size_t iter = 0;
115 std::size_t free_index = start_index + 1;
116
117 auto start_vert = add_to_tree(&start_tree, start, start_index, start_index, 0);
118
119 // Add goals to tree
120 std::vector<NNNode> goal_verts;
121 goal_verts.reserve(goals.size());
122
123 for (const auto &goal : goals)
124 {
125 goal_verts.emplace_back(add_to_tree(&goal_tree, goal, free_index, free_index, 0));
126 free_index++;
127 }
128
129 // trees
130 bool tree_a_is_start = not rrtc_settings.start_tree_first;
131 auto *tree_a = (rrtc_settings.start_tree_first) ? &goal_tree : &start_tree;
132 auto *tree_b = (rrtc_settings.start_tree_first) ? &start_tree : &goal_tree;
133
134 // Search loop
135 while (iter++ < rrtc_settings.max_iterations and free_index < rrtc_settings.max_samples)
136 {
137 float asize = tree_a->size();
138 float bsize = tree_b->size();
139 float ratio = std::abs(asize - bsize) / asize;
140
141 // Balanced RRTC
142 if ((not rrtc_settings.balance) or ratio < rrtc_settings.tree_ratio)
143 {
144 std::swap(tree_a, tree_b);
145 tree_a_is_start = not tree_a_is_start;
146 }
147
148 const auto temp = rng->next();
149
150 NNNode goal_vert = *std::min_element(
151 goal_verts.begin(),
152 goal_verts.end(),
153 [&temp](const auto &a, const auto &b)
154 { return temp.distance(a.array) < temp.distance(b.array); });
155
156 const auto &root_vert = tree_a_is_start ? start_vert : goal_vert;
157 const auto &target_vert = tree_a_is_start ? goal_vert : start_vert;
158
159 const float g_hat = temp.distance(root_vert.array);
160 const float h_hat = temp.distance(target_vert.array);
161 const float f_hat = g_hat + h_hat;
162
163 // The range between the minimum possible cost and maximum allowable cost
164 // - Floating point error can result in a (barely) negative range
165 // - If c_range is 0, only valid connection is to root of tree
166 // (sampled upper cost bound == g^)
167 const float c_range = std::max(max_cost - f_hat, 0.0F);
168
169 // Sampled upper cost bound
170 const float c_rand = (rng->dist.uniform_01() * c_range) + g_hat;
171
172 // Find nearest with asymmetric cost function
173 auto [nearest_node, nearest_distance] = find_nearest(tree_a, root_vert, temp, c_rand);
174 if (rrtc_settings.dynamic_domain and radii[nearest_node.index] < nearest_distance)
175 {
176 continue;
177 }
178
179 const auto nearest_vector = temp - nearest_node.array;
180 bool reach = nearest_distance < rrtc_settings.range;
181 const auto extension_vector =
182 (reach) ? nearest_vector : nearest_vector * (rrtc_settings.range / nearest_distance);
183
184 // Evaluate edge reaching towards sample
185 if (validate_vector<Robot, rake, resolution>(
186 nearest_node.array,
187 extension_vector,
188 (reach) ? nearest_distance : rrtc_settings.range,
189 environment))
190 {
191 const auto new_configuration = nearest_node.array + extension_vector;
192
193 // Calculate and store actual node cost
194 auto new_cost = nearest_node.cost + new_configuration.distance(nearest_node.array);
195
196 // If resampling costs to try and find a better parent...
197 if (settings.cost_bound_resample)
198 {
199 const float g_hat = new_configuration.distance(root_vert.array);
200
201 // Continuously resample cost until an invalid connection is found
202 for (auto i = 0U; i < settings.max_cost_bound_resamples; ++i)
203 {
204 const float c_range = std::max(new_cost - g_hat, 0.0F);
205 const float c_rand = (rng->dist.uniform_01() * c_range) + g_hat;
206
207 auto [new_nearest_node, new_nearest_distance] =
208 find_nearest(tree_a, root_vert, new_configuration, c_rand);
209
210 // If we have connected:
211 // to the same parent
212 // with a worse cost
213 // to the best possible parent before this (crange == 0 \equiv cost == g^)
214 // ...then stop spending effort resampling costs
215 if (new_nearest_node.index == nearest_node.index or
216 new_nearest_node.cost + new_nearest_distance >= new_cost or c_range == 0)
217 {
218 break;
219 }
220 // Validate edge to newly found parent
221 else if (validate_vector<Robot, rake, resolution>(
222 new_nearest_node.array,
223 new_configuration - new_nearest_node.array,
224 new_nearest_distance,
225 environment))
226 {
227 // Congratulations to the new parent
228 nearest_node = new_nearest_node;
229 new_cost = new_nearest_node.cost + new_nearest_distance;
230 }
231 // The edge is invalid, we have failed a connection. Stop resampling!
232 else
233 {
234 break;
235 }
236 }
237 }
238
239 add_to_tree(tree_a, new_configuration, free_index, nearest_node.index, new_cost);
240 free_index++;
241
242 if (rrtc_settings.dynamic_domain and
243 radii[nearest_node.index] != std::numeric_limits<float>::max())
244 {
245 radii[nearest_node.index] *= (1 + rrtc_settings.alpha);
246 }
247
248 // Extend to goal tree
249
250 // Because we are extending to the other tree, we need to change our upper cost bound
251 // We need to find a connection that improves upon our current best solution cost
252 // The cost from the root of the other tree to our new vertex, + our new vertex's cost
253 // through the current tree, must be lesser than our maximum path cost
254 // Therefore, our maximum allowable cost for a connection through the other tree is
255 // max_cost - vertex_cost
256 const auto [other_nearest_node, other_nearest_distance] =
257 find_nearest(tree_b, target_vert, new_configuration, max_cost - new_cost);
258 const auto other_nearest_vector = other_nearest_node.array - new_configuration;
259
260 // Just to be safe, make sure we've improved upon our best solution
261 if (new_cost + other_nearest_distance + other_nearest_node.cost >= max_cost)
262 {
263 continue;
264 }
265
266 // Extend incrementally towards other tree
267 const std::size_t n_extensions = std::ceil(other_nearest_distance / rrtc_settings.range);
268 const float increment_length = other_nearest_distance / static_cast<float>(n_extensions);
269 const auto increment = other_nearest_vector * (1.0F / static_cast<float>(n_extensions));
270
271 std::size_t i_extension = 0;
272 auto prior = new_configuration;
273 for (; i_extension < n_extensions and
274 validate_vector<Robot, rake, resolution>(
275 prior, increment, increment_length, environment) and
276 free_index < rrtc_settings.max_samples;
277 ++i_extension)
278 {
279 const auto next = prior + increment;
280 add_to_tree(
281 tree_a,
282 next,
283 free_index,
284 free_index - 1,
285 increment_length + costs[free_index - 1]);
286
287 free_index++;
288 prior = next;
289 }
290
291 if (i_extension == n_extensions) // connected
292 {
293 auto current = free_index - 1;
294 result.path.emplace_back(buffer_index(current));
295 while (parents[current] != current)
296 {
297 auto parent = parents[current];
298 result.path.emplace_back(buffer_index(parent));
299 result.cost += result.path[result.path.size() - 1].distance(
300 result.path[result.path.size() - 2]);
301 current = parent;
302 }
303
304 std::reverse(result.path.begin(), result.path.end());
305 current = other_nearest_node.index;
306
307 while (parents[current] != current)
308 {
309 auto parent = parents[current];
310 result.path.emplace_back(buffer_index(parent));
311 result.cost += result.path[result.path.size() - 1].distance(
312 result.path[result.path.size() - 2]);
313 current = parent;
314 }
315
316 if (not tree_a_is_start)
317 {
318 std::reverse(result.path.begin(), result.path.end());
319 }
320
321 break;
322 }
323 }
324 else if (rrtc_settings.dynamic_domain)
325 {
326 if (radii[nearest_node.index] == std::numeric_limits<float>::max())
327 {
328 radii[nearest_node.index] = rrtc_settings.radius;
329 }
330 else
331 {
332 radii[nearest_node.index] = std::max(
333 radii[nearest_node.index] * (1.F - rrtc_settings.alpha),
334 rrtc_settings.min_radius);
335 }
336 }
337 }
338
339 result.iterations = iter;
340 return result;
341 }

369 inline static auto solve(
370 const Configuration &start,
371 const std::vector<Configuration> &goals,
372 const collision::Environment<FloatVector<rake>> &environment,
373 const AORRTCSettings &settings_in,
374 typename RNG::Ptr rng) noexcept -> PlanningResult<Robot>
375 {
376 auto start_time = std::chrono::steady_clock::now();
377
378 // Update the settings for internal searches
379 AORRTCSettings settings = settings_in; // make a mutable copy
380 const std::size_t &max_samples = settings.max_samples;
381 const std::size_t &max_iterations = settings.max_iterations;
382
383 // Configure internal RRTC settings
384 RRTCSettings &rrtc_settings = settings.rrtc;
385 rrtc_settings.max_iterations = max_iterations;
386 rrtc_settings.max_samples = max_samples;
387
388 PlanningResult<Robot> result;
389 float best_path_cost = std::numeric_limits<float>::max();
390 std::size_t iters = 0;
391
392 do
393 {
394 // Find an initial solution
395 result = RRTC::solve(start, goals, environment, rrtc_settings, rng);
396 iters += result.iterations;
397 } while (result.path.empty() and iters < settings.max_iterations);
398
399 // Simplify solution if enabled
400 if (settings.simplify_intermediate and not result.path.empty())
401 {
402 result = simplify<Robot, rake, resolution>(result.path, environment, settings.simplify, rng);
403 }
404
405 // Exit early if trivial, unsolved, or not optimizing
406 if (not settings.optimize or result.path.empty() or result.path.size() == 2)
407 {
408 return result;
409 }
410
411 // We have a new best solution
412 PlanningResult<Robot> final_result;
413 final_result.path = result.path;
414 best_path_cost = result.path.cost();
415
416 float best_possible_cost = std::numeric_limits<float>::max();
417 for (const auto &goal : goals)
418 {
419 best_possible_cost = std::min(best_possible_cost, start.distance(goal));
420 }
421
422 ProlateHyperspheroid<Robot> phs(start, goals[0]);
423 phs.set_transverse_diameter(best_path_cost);
424
425 auto phs_rng = std::make_shared<ProlateHyperspheroidRNG<Robot>>(phs, rng);
426
427 AOX_RRTC instance(max_samples);
428
429 // If we get close to straight line, just call it.
430 // Also handles numerical issues with PHS when too close to straight line...
431 while (iters < max_iterations and (best_path_cost - best_possible_cost) > 1e-8)
432 {
433 // Update internal maximum iterations
434 rrtc_settings.max_iterations =
435 std::min(settings.max_iterations - iters, settings.max_internal_iterations);
436
437 // By default, use AORRTC
438 if (not settings.anytime)
439 {
440 // If there is a single goal, sample with PHS
441 if (settings.use_phs and goals.size() == 1)
442 {
443 result = instance.solve(start, goals, environment, settings, best_path_cost, phs_rng);
444 }
445 else
446 {
447 result = instance.solve(start, goals, environment, settings, best_path_cost, rng);
448 }
449 }
450 // If anytime, use Anytime RRTC
451 else
452 {
453 if (settings.use_phs and goals.size() == 1)
454 {
455 result = RRTC::solve(start, goals, environment, rrtc_settings, phs_rng);
456 }
457 else
458 {
459 result = RRTC::solve(start, goals, environment, rrtc_settings, rng);
460 }
461 }
462
463 iters += result.iterations;
464
465 // If last search found a solution
466 if (not result.path.empty())
467 {
468 // Simplify
469 if (settings.simplify_intermediate)
470 {
471 result = simplify<Robot, rake, resolution>(
472 result.path, environment, settings.simplify, rng);
473 }
474
475 // To be safe, ensure new path is actually a better solution
476 if (result.path.cost() < best_path_cost)
477 {
478 // Update best solution
479 final_result.path = result.path;
480 best_path_cost = result.path.cost();
481
482 phs_rng->phs.set_transverse_diameter(best_path_cost);
483 }
484 }
485 }
486
487 final_result.iterations = iters;
488 final_result.nanoseconds = vamp::utils::get_elapsed_nanoseconds(start_time);
489
490 return final_result;
491 }
