543 // converted from http://gruntthepeon.free.fr/ssemath/neon_mathfun.html
544 template <unsigned int = 0>
545 inline static constexpr auto sin(VectorT x) noexcept -> VectorT
546 {
547 using IntVector = SIMDVector<int32x4_t>;
548
549 // Constants
550 const auto c_cephes_FOPI = constant(1.27323954473516f); // 4 / M_PI
551 const auto c_minus_cephes_DP1 = constant(-0.78515625f);
552 const auto c_minus_cephes_DP2 = constant(-2.4187564849853515625e-4f);
553 const auto c_minus_cephes_DP3 = constant(-3.77489497744594108e-8f);
554 const auto c_sincof_p0 = constant(-1.9515295891E-4f);
555 const auto c_sincof_p1 = constant(8.3321608736E-3f);
556 const auto c_sincof_p2 = constant(-1.6666654611E-1f);
557 const auto one = constant(1.0f);
558 const auto half = constant(0.5f);
559
560 auto sign_mask_sin = cmp_less_than(x, zero_vector());
561 x = abs(x);
562
563 auto y = mul(x, c_cephes_FOPI);
564
565 auto emm2 = to<int32x4_t>(y);
566 emm2 = IntVector::add(emm2, IntVector::constant(1));
567 emm2 = IntVector::and_(emm2, IntVector::constant(~1));
568 y = from<int32x4_t>(emm2);
569
570 auto poly_mask = IntVector::and_(emm2, IntVector::constant(2));
571 poly_mask = IntVector::cmp_not_equal(poly_mask, IntVector::zero_vector());
572
573 auto xmm1 = mul(y, c_minus_cephes_DP1);
574 auto xmm2 = mul(y, c_minus_cephes_DP2);
575 auto xmm3 = mul(y, c_minus_cephes_DP3);
576 x = add(x, xmm1);
577 x = add(x, xmm2);
578 x = add(x, xmm3);
579
580 // Update sign mask
581 auto temp_mask = IntVector::and_(emm2, IntVector::constant(4));
582 temp_mask = IntVector::cmp_not_equal(temp_mask, IntVector::zero_vector());
583 sign_mask_sin = vreinterpretq_f32_u32(veorq_u32(
584 vreinterpretq_u32_f32(sign_mask_sin),
585 vreinterpretq_u32_f32(IntVector::template as<VectorT>(temp_mask))));
586
587 // Evaluate polynomials
588 auto z = mul(x, x);
589
590 // First polynomial (cosine)
591 auto y1 = mul(z, constant(2.443315711809948E-005f));
592 y1 = add(y1, constant(-1.388731625493765E-003f));
593 y1 = mul(y1, z);
594 y1 = add(y1, constant(4.166664568298827E-002f));
595 y1 = mul(y1, z);
596 y1 = mul(y1, z);
597 y1 = sub(y1, mul(z, half));
598 y1 = add(y1, one);
599
600 // Second polynomial (sine)
601 auto y2 = mul(z, c_sincof_p0);
602 y2 = add(y2, c_sincof_p1);
603 y2 = mul(y2, z);
604 y2 = add(y2, c_sincof_p2);
605 y2 = mul(y2, z);
606 y2 = mul(y2, x);
607 y2 = add(y2, x);
608
609 auto poly_mask_f = IntVector::template as<VectorT>(poly_mask);
610 auto ys = blend(y2, y1, poly_mask_f);
611 return blend(ys, neg(ys), sign_mask_sin);
612 }

614 // NOTE: Dummy parameter because otherwise we get constexpr errors with set1_ps...
615 template <unsigned int = 0>
616 inline static constexpr auto log(VectorT x) noexcept -> VectorT
617 {
618 using IntVector = SIMDVector<int32x4_t>;
619
620 const auto half = constant(0.5F);
621 const auto one = constant(1.0F);
622 auto invalid_mask = cmp_less_equal(x, zero_vector());
623
624 // cut off denormalized values
625 x = max(x, vreinterpretq_f32_u32(vdupq_n_u32(0x00800000u)));
626
627 auto emm0 = IntVector::shift_right(as<IntVector::VectorT>(x), 23);
628
629 x = and_(x, vreinterpretq_f32_u32(vdupq_n_u32(~0x7f800000u)));
630 x = or_(x, half);
631
632 // keep only the fractional part
633 emm0 = IntVector::sub(emm0, IntVector::constant(0x7f));
634 auto e = from<IntVector::VectorT>(emm0);
635
636 e = add(e, one);
637
638 // compute approx
639 auto mask = cmp_less_than(x, constant(0.707106781186547524f));
640 auto tmp = and_(x, mask);
641 x = sub(x, one);
642 e = sub(e, and_(one, mask));
643 x = add(x, tmp);
644
645 auto z = mul(x, x);
646

691 }
692 else if constexpr (blend_mask == 12)
693 {
694 return vbslq_f32(vcombine_u32(vcreate_u32(0l), vcreate_u32(0xffffffffffffffff)), b, a);

696 else if constexpr (blend_mask == 14)

698 return vbslq_f32(
699 vcombine_u32(vcreate_u32(0xffffffff00000000), vcreate_u32(0xffffffffffffffff)), b, a);

735 template <typename = void>
736 inline static auto gather(int32x4_t idxs, const ScalarT *base) noexcept -> VectorT
737 {
738 // Pretty sure there isn't a better way to do a 32-bit lookup table...
739 float32x4_t result = vdupq_n_f32(0);
740 result = vsetq_lane_f32(base[vgetq_lane_s32(idxs, 0)], result, 0);
741 result = vsetq_lane_f32(base[vgetq_lane_s32(idxs, 1)], result, 1);
742 result = vsetq_lane_f32(base[vgetq_lane_s32(idxs, 2)], result, 2);
743 result = vsetq_lane_f32(base[vgetq_lane_s32(idxs, 3)], result, 3);
744 return result;
745 }
