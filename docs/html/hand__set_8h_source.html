63// The hash and equality functions below are necessary for boost's unordered
64// set.
65namespace boost {
66template <>
67struct hash<Eigen::Vector3i> {
68 inline size_t operator()(Eigen::Vector3i const &v) const {
69 std::size_t seed = 0;
70
71 for (int i = 0; i < v.size(); i++) {
72 boost::hash_combine(seed, v(i));
73 }
74
75 return seed;
76 }
77};
78} // namespace boost

83struct Vector3iEqual {
84 inline bool operator()(const Eigen::Vector3i &a,
85 const Eigen::Vector3i &b) const {
86 return a(0) == b(0) && a(1) == b(1) && a(2) == b(2);
87 }
88};
