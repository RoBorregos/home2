31inline long long get_us() {
32#ifdef _WIN32
33#define EPOCHFILETIME (116444736000000000UL)
34 FILETIME ft;
35 LARGE_INTEGER li;
36 GetSystemTimeAsFileTime(&ft);
37 li.LowPart = ft.dwLowDateTime;
38 li.HighPart = ft.dwHighDateTime;
39 return (li.QuadPart - EPOCHFILETIME) / 10;
40#else
41 struct timespec t;
42 clock_gettime(CLOCK_MONOTONIC, &t);
43 return (t.tv_sec * 1000000000 + t.tv_nsec) * 0.001;
44#endif
45}
