63 {
64 // Configuration space distance + Cost space distance

214 auto remove(const _T &data) noexcept -> bool
215 {
216 if (size_ == 0U)
217 {
218 return false;
219 }
220
221 NearQueue nbhQueue;
222
223 // find data in tree
224 const bool isPivot = nearestKInternal(data, 1, nbhQueue);
225 const _T *d = nbhQueue.top().second;
226
227 if (*d != data)
228 {
229 return false;
230 }
231
232 removed_.insert(d);
233 size_--;
234
235 // if we removed a pivot or if the capacity of removed elements
236 // has been reached, we rebuild the entire GNAT
237 if (isPivot or removed_.size() >= removedCacheSize_)
238 {
239 rebuildDataStructure();
240 }
241
242 return true;
243 }

314 auto nearestKInternal(const _T &data, std::size_t k, NearQueue &nbhQueue) const noexcept -> bool
315 {
316 NodeDist nodeDist;
317 NodeQueue nodeQueue;
318
319 const float dist = _T::distance(data, tree_->pivot_);
320 bool isPivot = tree_->insertNeighborK(nbhQueue, k, tree_->pivot_, data, dist);
321 tree_->nearestK(*this, data, k, nbhQueue, nodeQueue, isPivot);
322
323 while (not nodeQueue.empty())
324 {
325 dist = nbhQueue.top().first; // note the difference with nearestRInternal
326 nodeDist = nodeQueue.top();
327 nodeQueue.pop();
328
329 if (nbhQueue.size() == k and (nodeDist.second > nodeDist.first->maxRadius_ + dist or
330 nodeDist.second < nodeDist.first->minRadius_ - dist))
331 {
332 continue;
333 }
334
335 nodeDist.first->nearestK(*this, data, k, nbhQueue, nodeQueue, isPivot);
336 }
337
338 return isPivot;
339 }

342 void nearestRInternal(const _T &data, float radius, NearQueue &nbhQueue) const noexcept
343 {
344 const float dist = radius; // note the difference with nearestKInternal
345 NodeQueue nodeQueue;
346 NodeDist nodeDist;
347
348 tree_->insertNeighborR(nbhQueue, radius, tree_->pivot_, _T::distance(data, tree_->pivot_));
349 tree_->nearestR(*this, data, radius, nbhQueue, nodeQueue);
350
351 while (not nodeQueue.empty())
352 {
353 nodeDist = nodeQueue.top();
354 nodeQueue.pop();
355 if (nodeDist.second > nodeDist.first->maxRadius_ + dist or
356 nodeDist.second < nodeDist.first->minRadius_ - dist)
357 {
358 continue;
359 }
360
361 nodeDist.first->nearestR(*this, data, radius, nbhQueue, nodeQueue);
362 }

376 void kcenters(
377 const std::vector<_T> &data,
378 std::size_t k,
379 std::vector<std::size_t> &centers,
380 Eigen::MatrixXf &dists) noexcept
381 {
382 // array containing the minimum distance between each data point
383 // and the centers computed so far
384 std::vector<float> minDist(data.size(), std::numeric_limits<float>::infinity());
385 centers.clear();
386 centers.reserve(k);
387
388 if (static_cast<std::size_t>(dists.rows()) < data.size() or
389 static_cast<std::size_t>(dists.cols()) < k)
390 {
391 dists.resize(std::max(2U * (std::size_t)dists.rows() + 1U, data.size()), k);
392 }
393 // first center is picked randomly
394
395 centers.emplace_back(rng_.uniform_integer(0, static_cast<int>(data.size()) - 1));
396
397 for (auto i = 1U; i < k; ++i)
398 {
399 std::size_t ind = 0;
400 const _T &center = data[centers[i - 1]];
401 float maxDist = -std::numeric_limits<float>::infinity();
402
403 for (auto j = 0U; j < data.size(); ++j)
404 {
405 if ((dists(j, i - 1) = _T::distance(data[j], center)) < minDist[j])
406 {
407 minDist[j] = dists(j, i - 1);
408 }
409
410 // the j-th center is the one furthest away from center 0,..,j-1
411 if (minDist[j] > maxDist)
412 {
413 ind = j;
414 maxDist = minDist[j];
415 }
416 }
417
418 // no more centers available
419 if (maxDist < std::numeric_limits<float>::epsilon())
420 {
421 break;
422 }
423
424 centers.emplace_back(ind);
425 }
426
427 const _T &center = data[centers.back()];
428 const auto i = centers.size() - 1;
429
430 for (auto j = 0U; j < data.size(); ++j)
431 {
432 dists(j, i) = _T::distance(data[j], center);
433 }
434 }

437 class Node
438 {
439 public:
442 Node(std::size_t degree, std::size_t capacity, _T pivot)
443 : degree_(degree)
444 , pivot_(std::move(pivot))
445 , minRadius_(std::numeric_limits<float>::infinity())
446 , maxRadius_(-minRadius_)
447 , minRange_(degree, minRadius_)
448 , maxRange_(degree, maxRadius_)
449 {
450 // The "+1" is needed because we add an element before we check whether to split
451 data_.reserve(capacity + 1);
452 }
453
454 ~Node()
455 {
456 for (auto &child : children_)
457 {
458 delete child;
459 }
460 }
461
464 void updateRadius(float dist)
465 {
466 if (minRadius_ > dist)
467 {
468 minRadius_ = dist;
469 }
470
471 if (maxRadius_ < dist)
472 {
473 maxRadius_ = dist;
474 }
475 }
476
480 void updateRange(std::size_t i, float dist)
481 {
482 if (minRange_[i] > dist)
483 {
484 minRange_[i] = dist;
485 }
486
487 if (maxRange_[i] < dist)
488 {
489 maxRange_[i] = dist;
490 }
491 }
492
494 void add(GNAT &gnat, const _T &data)
495 {
496 if (children_.empty())
497 {
498 data_.emplace_back(data);
499 gnat.size_++;
500 if (needToSplit(gnat))
501 {
502 if (not gnat.removed_.empty())
503 {
504 gnat.rebuildDataStructure();
505 }
506 else if (gnat.size_ >= gnat.rebuildSize_)
507 {
508 gnat.rebuildSize_ <<= 1;
509 gnat.rebuildDataStructure();
510 }
511 else
512 {
513 split(gnat);
514 }
515 }
516 }
517 else
518 {
519 std::vector<float> dist(children_.size());
520 float minDist = dist[0] = _T::distance(data, children_[0]->pivot_);
521 std::size_t minInd = 0;
522
523 for (auto i = 1U; i < children_.size(); ++i)
524 {
525 if ((dist[i] = _T::distance(data, children_[i]->pivot_)) < minDist)
526 {
527 minDist = dist[i];
528 minInd = i;
529 }
530 }
531
532 for (auto i = 0U; i < children_.size(); ++i)
533 {
534 children_[i]->updateRange(minInd, dist[i]);
535 }
536
537 children_[minInd]->updateRadius(minDist);
538 children_[minInd]->add(gnat, data);
539 }

543 auto needToSplit(const GNAT &gnat) const noexcept -> bool
544 {
545 const auto sz = data_.size();
546 return sz > gnat.maxNumPtsPerLeaf_ and sz > degree_;
547 }
548
552 void split(GNAT &gnat)
553 {
554 Eigen::MatrixXf dists(data_.size(), degree_);
555 std::vector<std::size_t> pivots;
556
557 children_.reserve(degree_);
558 gnat.kcenters(data_, degree_, pivots, dists);
559 for (const auto &pivot : pivots)
560 {
561 children_.emplace_back(new Node(degree_, gnat.maxNumPtsPerLeaf_, data_[pivot]));
562 }
563
564 degree_ = pivots.size(); // in case fewer than degree_ pivots were found
565
566 for (auto j = 0U; j < data_.size(); ++j)
567 {
568 std::size_t k = 0;
569 for (auto i = 1U; i < degree_; ++i)
570 {
571 if (dists(j, i) < dists(j, k))
572 {
573 k = i;
574 }
575 }
576
577 auto *child = children_[k];
578 if (j != pivots[k])
579 {
580 child->data_.emplace_back(data_[j]);
581 child->updateRadius(dists(j, k));
582 }
583
584 for (auto i = 0U; i < degree_; ++i)
585 {
586 children_[i]->updateRange(k, dists(j, i));
587 }
588 }
589
590 for (auto &child : children_)
591 {
592 // make sure degree lies between minDegree_ and maxDegree_
593 child->degree_ = std::min(
594 std::max(
595 static_cast<std::size_t>((degree_ * child->data_.size()) / data_.size()),
596 gnat.minDegree_),
597 gnat.maxDegree_);
598
599 // singleton
600 if (child->minRadius_ >= std::numeric_limits<float>::infinity())
601 {
602 child->minRadius_ = child->maxRadius_ = 0.;
603 }
604 }
605
606 // this does more than clear(); it also sets capacity to 0 and frees the memory
607 std::vector<_T> tmp;
608 data_.swap(tmp);
609
610 // check if new leaves need to be split
611 for (auto &child : children_)
612 {
613 if (child->needToSplit(gnat))
614 {
615 child->split(gnat);
616 }
617 }
618 }

621 auto insertNeighborK(NearQueue &nbh, std::size_t k, const _T &data, const _T &key, float dist)
622 const noexcept -> bool
623 {
624 if (nbh.size() < k)
625 {
626 nbh.emplace(dist, &data);
627 return true;
628 }
629
630 if (dist < nbh.top().first or (dist < std::numeric_limits<float>::epsilon() and data == key))
631 {
632 nbh.pop();
633 nbh.emplace(dist, &data);
634 return true;
635 }
636
637 return false;
638 }

645 void nearestK(
646 const GNAT &gnat,
647 const _T &data,
648 std::size_t k,
649 NearQueue &nbh,
650 NodeQueue &nodeQueue,
651 bool &isPivot) const noexcept
652 {
653 for (const auto &d : data_)
654 {
655 if (not gnat.isRemoved(d))
656 {
657 if (insertNeighborK(nbh, k, d, data, _T::distance(data, d)))
658 {
659 isPivot = false;
660 }
661 }
662 }
663
664 if (!children_.empty())
665 {
666 std::size_t sz = children_.size();
667 std::size_t offset = gnat.offset_++;
668 std::vector<float> distToPivot(sz);
669 std::vector<int> permutation(sz);
670
671 for (auto i = 0U; i < sz; ++i)
672 {
673 permutation[i] = (i + offset) % sz;
674 }
675
676 for (auto i = 0U; i < sz; ++i)
677 {
678 if (permutation[i] >= 0)
679 {
680 const auto &child = children_[permutation[i]];
681 distToPivot[permutation[i]] = _T::distance(data, child->pivot_);
682 if (insertNeighborK(nbh, k, child->pivot_, data, distToPivot[permutation[i]]))
683 {
684 isPivot = true;
685 }
686
687 if (nbh.size() == k)
688 {
689 const auto dist = nbh.top().first; // note difference with nearestR
690 for (auto j = 0U; j < sz; ++j)
691 {
692 if (permutation[j] >= 0 and i != j and
693 (distToPivot[permutation[i]] - dist >
694 child->maxRange_[permutation[j]] or
695 distToPivot[permutation[i]] + dist <
696 child->minRange_[permutation[j]]))
697 {
698 permutation[j] = -1;
699 }
700 }
701 }
702 }
703 }
704
705 const auto dist = nbh.top().first;
706 for (auto p : permutation)
707 {
708 if (p >= 0)
709 {
710 const auto &child = children_[p];
711 if (nbh.size() < k or (distToPivot[p] - dist <= child->maxRadius_ and
712 distToPivot[p] + dist >= child->minRadius_))
713 {
714 nodeQueue.emplace(child, distToPivot[p]);
715 }
716 }
717 }
718 }

730 void nearestR(const GNAT &gnat, const _T &data, float r, NearQueue &nbh, NodeQueue &nodeQueue)
731 const noexcept
732 {
733 float dist = r; // note difference with nearestK
734
735 for (const auto &d : data_)
736 {
737 if (not gnat.isRemoved(d))
738 {
739 insertNeighborR(nbh, r, d, _T::distance(data, d));
740 }
741 }
742
743 if (not children_.empty())
744 {
745 std::size_t sz = children_.size();
746 std::size_t offset = gnat.offset_++;
747 std::vector<float> distToPivot(sz);
748 std::vector<int> permutation(sz);
749
750 // Not a random permutation, but processing the children in slightly different order is
751 // "good enough" to get a performance boost. A call to std::shuffle takes too long.
752 for (auto i = 0U; i < sz; ++i)
753 {
754 permutation[i] = (i + offset) % sz;
755 }
756
757 for (auto i = 0U; i < sz; ++i)
758 {
759 if (permutation[i] >= 0)
760 {
761 const auto &child = children_[permutation[i]];
762 distToPivot[permutation[i]] = _T::distance(data, child->pivot_);
763 insertNeighborR(nbh, r, child->pivot_, distToPivot[permutation[i]]);
764
765 for (auto j = 0U; j < sz; ++j)
766 {
767 if (permutation[j] >= 0 and i != j and
768 (distToPivot[permutation[i]] - dist > child->maxRange_[permutation[j]] or
769 distToPivot[permutation[i]] + dist < child->minRange_[permutation[j]]))
770 {
771 permutation[j] = -1;
772 }
773 }
774 }
775 }
776
777 for (const auto &p : permutation)
778 {
779 if (p >= 0)
780 {
781 const auto &child = children_[p];
782
783 if (distToPivot[p] - dist <= child->maxRadius_ and
784 distToPivot[p] + dist >= child->minRadius_)
785 {
786 nodeQueue.emplace(child, distToPivot[p]);
787 }
788 }
789 }
790 }
