454 // converted from http://gruntthepeon.free.fr/ssemath/
455 template <unsigned int = 0>
456 inline static constexpr auto sin(VectorT x) noexcept -> VectorT
457 {
458 using IntVector = SIMDVector<__m256i>;
459
460 const auto ps_sign_mask = constant_int(0x80000000);
461 const auto ps_cephes_FOPI = constant(1.27323954473516f); // 4/Pi
462 const auto pi32_1 = IntVector::constant(1);
463 const auto pi32_inv1 = IntVector::constant(~1);
464 const auto pi32_4 = IntVector::constant(4);
465 const auto pi32_2 = IntVector::constant(2);
466 const auto ps_minus_cephes_DP1 = constant(-0.78515625f);
467 const auto ps_minus_cephes_DP2 = constant(-2.4187564849853515625e-4f);
468 const auto ps_minus_cephes_DP3 = constant(-3.77489497744594108e-8f);
469 const auto ps_coscof_p0 = constant(2.443315711809948E-005f);
470 const auto ps_coscof_p1 = constant(-1.388731625493765E-003f);
471 const auto ps_coscof_p2 = constant(4.166664568298827E-002f);
472 const auto ps_0p5 = constant(0.5f);
473 const auto ps_1 = constant(1.0f);
474 const auto ps_sincof_p0 = constant(-1.9515295891E-4f);
475 const auto ps_sincof_p1 = constant(8.3321608736E-3f);
476 const auto ps_sincof_p2 = constant(-1.6666654611E-1f);
477
478 auto sign_bit = x;
479
480 x = abs(x);
481 sign_bit = and_(sign_bit, ps_sign_mask);
482 auto y = mul(x, ps_cephes_FOPI);
483
484 auto emm2 = IntVector::from(y);
485
486 // j=(j+1) & (~1) (see the cephes sources)
487 emm2 = IntVector::add(emm2, pi32_1);
488 emm2 = IntVector::and_(emm2, pi32_inv1);
489 y = from<IntVector::VectorT>(emm2);
490
491 // Get the swap sign flag
492 auto emm0 = IntVector::and_(emm2, pi32_4);
493 emm0 = IntVector::shift_left(emm0, 29);
494
495 // Get the polynom selection mask
496 // There is one polynom for 0 <= x <= Pi/4
497 // and another one for Pi/4 < x <= Pi/2
498 // Both branches will be computed.
499 emm2 = IntVector::and_(emm2, pi32_2);
500 emm2 = IntVector::cmp_equal(emm2, IntVector::zero_vector());
501
502 auto swap_sign_bit = IntVector::template as<VectorT>(emm0);
503 auto poly_mask = IntVector::template as<VectorT>(emm2);
504 sign_bit = _mm256_xor_ps(sign_bit, swap_sign_bit);
505
506 // The magic pass: "Extended precision modular arithmetic"
507 // x = ((x - y * DP1) - y * DP2) - y * DP3;
508 auto xmm1 = mul(y, ps_minus_cephes_DP1);
509 auto xmm2 = mul(y, ps_minus_cephes_DP2);
510 auto xmm3 = mul(y, ps_minus_cephes_DP3);
511 x = add(x, xmm1);
512 x = add(x, xmm2);
513 x = add(x, xmm3);
514
515 // Evaluate the first polynom (0 <= x <= Pi/4)
516 y = ps_coscof_p0;
517 auto z = mul(x, x);
518 y = mul(y, z);
519 y = add(y, ps_coscof_p1);
520 y = mul(y, z);
521 y = add(y, ps_coscof_p2);
522 y = mul(y, z);
523 y = mul(y, z);
524 auto tmp = mul(z, ps_0p5);
525 y = sub(y, tmp);
526 y = add(y, ps_1);
527
528 // Evaluate the second polynom (Pi/4 <= x <= 0)
529 auto y2 = ps_sincof_p0;
530 y2 = mul(y2, z);
531 y2 = add(y2, ps_sincof_p1);
532 y2 = mul(y2, z);
533 y2 = add(y2, ps_sincof_p2);
534 y2 = mul(y2, z);
535 y2 = mul(y2, x);
536 y2 = add(y2, x);
537
538 // Select the correct result from the two polynoms
539 xmm3 = poly_mask;
540 y2 = and_(xmm3, y2);
541 y = _mm256_andnot_ps(xmm3, y);
542 y = add(y, y2);
543
544 // Update the sign
545 y = _mm256_xor_ps(y, sign_bit);
546
547 return y;
548 }

552 inline static constexpr auto log(VectorT x) noexcept -> VectorT
553 {
554 using IntVector = SIMDVector<__m256i>;
555
556 const auto half = constant(0.5F);
557 const auto one = constant(1.0F);
558 auto invalid_mask = cmp_less_equal(x, zero_vector());
559
560 // cut off denormalized values
561 x = max(x, constant_int(0x00800000));
562
563 auto emm0 = IntVector::shift_right(as<IntVector::VectorT>(x), 23);
564
565 x = and_(x, constant_int(~0x7f800000));
566 x = or_(x, half);
567
568 // keep only the fractional part
569 emm0 = IntVector::sub(emm0, IntVector::constant(0x7f));
570 auto e = from<IntVector::VectorT>(emm0);
571
572 e = add(e, one);
573
574 // compute approx
575 auto mask = cmp_less_than(x, constant(0.707106781186547524f));
576 auto tmp = and_(x, mask);
577 x = sub(x, one);
578 e = sub(e, and_(one, mask));
579 x = add(x, tmp);
580
581 auto z = mul(x, x);
582
583 auto y = constant(7.0376836292E-2f);
584 y = mul(y, x);
585 y = add(y, constant(-1.1514610310E-1f));
586 y = mul(y, x);
587 y = add(y, constant(1.1676998740E-1f));
588 y = mul(y, x);
589 y = add(y, constant(-1.2420140846E-1f));
590 y = mul(y, x);
591 y = add(y, constant(+1.4249322787E-1f));
592 y = mul(y, x);
593 y = add(y, constant(-1.6668057665E-1f));
594 y = mul(y, x);
595 y = add(y, constant(+2.0000714765E-1f));
596 y = mul(y, x);
597 y = add(y, constant(-2.4999993993E-1f));
598 y = mul(y, x);
599 y = add(y, constant(+3.3333331174E-1f));
600 y = mul(y, mul(x, z));
601 tmp = mul(e, constant(-2.12194440e-4f));
602 y = add(y, tmp);
603 tmp = mul(z, half);
604 y = sub(y, tmp);
605 tmp = mul(e, constant(0.693359375f));
606 x = add(x, add(y, tmp));
607
608 x = or_(x, invalid_mask); // negative arg will be NAN
609 return x;
610 }
