27 // A cuboid, or rectangular prism, shape oriented in an arbitrary frame. The
28 // cube is defined by its center (x, y, z), the three axes defining its frame
29 // (axis_1_x, axis_1_y, axis_1_z, etc.), and the half-lengths along each axis
30 // (axis_1_r, etc.)
31 template <typename DataT>
32 struct Cuboid : public Shape<DataT>
33 {
34 DataT x;
35 DataT y;
36 DataT z;

52 inline constexpr auto compute_min_distance() -> DataT
53 {
54 auto d_1 = dot_3(-x, -y, -z, axis_1_x, axis_1_y, axis_1_z);
55 auto d_2 = dot_3(-x, -y, -z, axis_2_x, axis_2_y, axis_2_z);
56 auto d_3 = dot_3(-x, -y, -z, axis_3_x, axis_3_y, axis_3_z);
57
58 auto v_1 = clamp(d_1, -axis_1_r, axis_1_r);
59 auto v_2 = clamp(d_2, -axis_2_r, axis_2_r);
60 auto v_3 = clamp(d_3, -axis_3_r, axis_3_r);
61
62 auto x_n = x + axis_1_x * v_1 + axis_2_x * v_2 + axis_3_x * v_3;
63 auto y_n = y + axis_1_y * v_1 + axis_2_y * v_2 + axis_3_y * v_3;
64 auto z_n = z + axis_1_z * v_1 + axis_2_z * v_2 + axis_3_z * v_3;
65
66 return sqrt(x_n * x_n + y_n * y_n + z_n * z_n);
67 }

129 // A cylinder shape oriented in an arbitrary frame. The cylinder is defined by
130 // two endpoints, which we store as the first point (x1, y1, z1) and the vector
131 // from the first point to the second point (xv, yv, zv), and its radius (r)
132 template <typename DataT>
133 struct Cylinder : public Shape<DataT>
134 {
135 DataT x1;
136 DataT y1;
137 DataT z1;
138
139 DataT xv;
140 DataT yv;
141 DataT zv;
142
143 DataT r;
144
145 // This is the reciprocal of the dot product of the vector between the
146 // cylinder's endpoints, i.e. rdv = 1 / ([xv yv zv] . [xv yv zv]) We store it
147 // for convenience to make some computations faster
148 DataT rdv;
149
150 inline constexpr auto x2() -> DataT
151 {
152 return x1 + xv;
153 }
154
155 inline constexpr auto y2() -> DataT
156 {
157 return y1 + yv;
158 }
159
160 inline constexpr auto z2() -> DataT
161 {
162 return z1 + zv;
163 }
164
165 inline constexpr auto compute_min_distance() -> DataT
166 {
167 auto dot = clamp(dot_3(-x1, -y1, -z1, xv, yv, zv) * rdv, 0.F, 1.F);
168
169 auto xp = x1 + xv * dot;
170 auto yp = y1 + yv * dot;
171 auto zp = z1 + zv * dot;
172
173 auto xo = -xp;
174 auto yo = -yp;
175 auto zo = -zp;
176
177 auto ol = sqrt(dot_3(xo, yo, zo, xo, yo, zo));
178 xo = xo / ol;
179 yo = yo / ol;
180 zo = zo / ol;
181
182 auto ro = clamp(ol, 0.F, r);
183
184 auto xn = xp + ro * xo;
185 auto yn = yp + ro * yo;
186 auto zn = zp + ro * zo;
187
188 return sqrt(xn * xn + yn * yn + zn * zn);
189 }
