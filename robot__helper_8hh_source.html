234 inline static auto fk(const Type &c_in) -> std::vector<vamp::collision::Sphere<float>>
235 {
236 typename Robot::template Spheres<1> out;
237 Robot::template sphere_fk<1>(Input::template block<1>(c_in), out);
238
239 std::vector<vamp::collision::Sphere<float>> result(Robot::n_spheres);
240 for (auto i = 0U; i < Robot::n_spheres; ++i)

242 result[i] = vamp::collision::Sphere<float>{
243 out.x[{i, 0}], out.y[{i, 0}], out.z[{i, 0}], out.r[{i, 0}]};
244 }
245
246 return result;

255 inline static auto
256 validate(const Type &c_in, const EnvironmentInput &environment, bool check_bounds = false) -> bool
257 {
258 auto configuration = Input::to(c_in);
259 auto copy = configuration.trim();
260 Robot::descale_configuration(copy);
261
262 const bool in_bounds = (copy <= 1.F).all() and (copy >= 0.F).all();
263
264 return (not check_bounds or in_bounds) and
265 vamp::planning::validate_motion<Robot, rake, 1>(
266 configuration, configuration, EnvironmentVector(environment));
267 }

284 inline static auto filter_self_from_pointcloud(
285 const std::vector<collision::Point> &pc,
286 float point_radius,
287 const Type &c_in,
288 const EnvironmentInput &environment) -> std::vector<collision::Point>
289 {
290 // TODO: Do this smarter with SIMD CC
291 EnvironmentVector ev(environment);
292
293 typename Robot::template Spheres<1> out;
294 Robot::template sphere_fk<1>(Input::template block<1>(c_in), out);
295
296 std::vector<collision::Point> filtered;
297 filtered.reserve(pc.size());
298
299 for (const auto &point : pc)
300 {
301 const float x = point[0], y = point[1], z = point[2], r = point_radius;
302
303 bool valid = true;
304 for (auto i = 0U; i < Robot::n_spheres; ++i)
305 {
306 if (collision::sphere_sphere_sql2(
307 out.x[{i, 0}], out.y[{i, 0}], out.z[{i, 0}], out.r[{i, 0}], x, y, z, r) < 0 or
308 sphere_environment_in_collision<>(ev, x, y, z, r))
309 {
310 valid = false;
311 break;
312 }
313 }
314
315 if (valid)
316 {
317 filtered.emplace_back(point);
318 }
319 }
320
321 return filtered;
322 }
