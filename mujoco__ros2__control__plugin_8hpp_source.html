279 // Weld-on-attach: subscribes to /attached_collision_object (what
280 // MoveIt publishes when pick_server attaches the pick object to
281 // link_eef) and toggles a named MuJoCo weld equality so the object
282 // is physically pinned to the gripper instead of depending on
283 // fragile contact friction. Detach publishes REMOVE and we turn
284 // the weld off again. The activation is deferred via a timer so
285 // the gripper finishes physically closing on the object first;
286 // otherwise the weld freezes the cube at whatever offset it had
287 // at attach time (before close_gripper), and it hangs skewed on
288 // the gripper through the retreat.
289 rclcpp::Subscription<moveit_msgs::msg::AttachedCollisionObject>::SharedPtr attach_sub_;
290 // Gripper trajectory listener: syncs adhesion actuators with the
291 // finger close/open command so adhesion releases the object at the
292 // exact moment the gripper opens (which is the ground truth for
293 // "let go"), regardless of whether MoveIt publishes a detach.
294 rclcpp::Subscription<trajectory_msgs::msg::JointTrajectory>::SharedPtr gripper_traj_sub_;
295 rclcpp::TimerBase::SharedPtr weld_activation_timer_;
296 int pending_weld_eq_id_{-1};
