37 inline auto contained_by_internal_ball(const Point &p, float r) const noexcept -> float
38 {
39 assert(distsq_to(p) <= std::numeric_limits<float>::epsilon());
40 const float d0 = std::max(p[0] - lower[0], upper[0] - p[0]);
41 const float d1 = std::max(p[1] - lower[1], upper[1] - p[1]);
42 const float d2 = std::max(p[2] - lower[2], upper[2] - p[2]);
43 return (d0 * d0 + d1 * d1 + d2 * d2) <= r;
44 }
45
46 inline auto distsq_to(const Point &point) const noexcept -> float
47 {
48 const float d0 = point[0] - std::clamp(point[0], lower[0], upper[0]);
49 const float d1 = point[1] - std::clamp(point[1], lower[1], upper[1]);
50 const float d2 = point[2] - std::clamp(point[2], lower[2], upper[2]);
51
52 return d0 * d0 + d1 * d1 + d2 * d2;
53 }
54
55 inline auto affords(const Point &point, const float max_affordance_l2) const noexcept -> bool
56 {
57 return distsq_to(point) <= max_affordance_l2;
58 }

367 // Test whether a sphere centered at `center` with radius-squared `radius_sq` collides with any
368 // point in this tree.
369 // Returns `true` if in collision and `false` if not.
370 [[nodiscard]] auto collides(const Point &center, float r) const noexcept -> bool
371 {
372 if (aabb_top.distsq_to(center) > r * r)
373 {
374 return false;
375 }
376
377 std::size_t test_idx = 0;
378 for (uint8_t i = 0, k = 0; i < nlog2; i++)
379 {
380 test_idx = 2 * test_idx + 1 + (center[k] >= tests[test_idx]);
381 k = (k + 1) % 3;
382 }
383
384 const std::size_t z = test_idx - tests.size();
385
386 r += r_point;
387 const float radius_sq = r * r;
388 if (aabbs[z].distsq_to(center) > radius_sq)
389 {
390 return false;
391 }
392
393 const uint32_t start = aff_starts[z];
394 const uint32_t end = aff_starts[z + 1];
395
396 const auto xc = FVectorT::fill(center[0]);
397 const auto yc = FVectorT::fill(center[1]);
398 const auto zc = FVectorT::fill(center[2]);
399 const auto rc = FVectorT::fill(radius_sq);
400 for (uint32_t i = start; i < end; i++)
401 {
402 const auto distsq =
403 sql2_3(affordances[0][i], affordances[1][i], affordances[2][i], xc, yc, zc);
404 if (distsq.test_any_less_equal(rc))
405 {
406 return true;
407 }
408 }
409
410 return false;
411 }

413 // Determine whether any of a set of spheres collides with a point in this tree.
414 //
415 // Templates
416 //
417 // - `FVectorT`: type of a vector of floats
418 // - `IVectorT`: type of a vector of integer indexes
419 //
420 // Inputs
421 //
422 // - `centers`: (x, y, z) struct-of-arrays of the centers of each sphere.
423 // - `radii`: SIMD vector of the radii of each sphere.
424 auto collides_simd(const std::array<FVectorT, 3> &centers, FVectorT radii) const noexcept -> bool
425 {
426 // Test against top AABB
427 FVectorT inbounds =
428 (centers[0] + radii >= aabb_top.lower[0]) & (centers[0] - radii <= aabb_top.upper[0]);
429
430 for (uint8_t k = 1; k < 3; k++)
431 {
432 inbounds = inbounds & (centers[k] + radii >= aabb_top.lower[k]) &
433 (centers[k] - radii <= aabb_top.upper[k]);
434 }
435
436 if (inbounds.none())
437 {
438 return false;
439 }
440
441 FVectorT these_tests = FVectorT::fill(tests[0]);
442 FVectorT cmp_results = centers[0].greater_equal(these_tests);
443 auto idxs = (cmp_results >> 31U).template as<IVectorT>() + 1;
444
445 // Search downward through the tree, parallel across each point
446 for (uint8_t i = 1, k = 1; i < nlog2; i++)
447 {
448 these_tests = FVectorT::gather(tests.data(), idxs);
449 cmp_results = centers[k].greater_equal(these_tests);
450 idxs = (idxs << 1U) + (cmp_results >> 31U).template as<IVectorT>() + 1;
451 k = (k + 1) % 3;
452 }
453
454 const IVectorT zs = idxs - tests.size();
455
456 // Test whether points are in the AABBs
457 // NOTE: Now is when we need to add r_point, since these AABBs are really "point volume AABBs" -
458 // we can't just test if the query is in the AABB, but rather if it's in the AABB when fattened by
459 // the radius of the points the AABB contains
460 radii = radii + r_point;
461 IVectorT zs6 = zs * 6;
462 const float *const aabb_ptr = &aabbs.front().lower.front();
463
464 const auto rc_sq = radii * radii;
465
466 auto d0 = centers[0] -
467 centers[0].clamp(FVectorT::gather(aabb_ptr, zs6), FVectorT::gather(aabb_ptr, zs6 + 3));
468 auto d1 =
469 centers[1] -
470 centers[1].clamp(FVectorT::gather(aabb_ptr, zs6 + 1), FVectorT::gather(aabb_ptr, zs6 + 4));
471 auto d2 =
472 centers[2] -
473 centers[2].clamp(FVectorT::gather(aabb_ptr, zs6 + 2), FVectorT::gather(aabb_ptr, zs6 + 5));
474
475 auto distsq_to = d0 * d0 + d1 * d1 + d2 * d2;
476 inbounds = inbounds & (distsq_to <= rc_sq);
477 if (inbounds.none())
478 {
479 return false;
480 }
481
482 // Convert the terminal test indices to reference indices for the affordance buffer
483 const auto *affdata = reinterpret_cast<const int32_t *>(aff_starts.data());
484 const IVectorT starts_v = IVectorT::gather(affdata, zs);
485 IVectorT ends_v = inbounds.template as<IVectorT>() & IVectorT::gather(affdata, zs + 1);
486
487 const auto starts = starts_v.to_array();
488 const auto ends = ends_v.to_array();
489
490 for (uint8_t j = 0; j < FVectorT::num_scalars; j++)
491 {
492 const auto xc = centers[0].broadcast(j);
493 const auto yc = centers[1].broadcast(j);
494 const auto zc = centers[2].broadcast(j);
495 const auto rc = rc_sq.broadcast(j);
496 for (auto i = starts[j]; i < ends[j]; ++i)
497 {
498 const auto distsq =
499 sql2_3(affordances[0][i], affordances[1][i], affordances[2][i], xc, yc, zc);
500 if (distsq.test_any_less_equal(rc))
501 {
502 return true;
503 }
504 }
505 }
506
507 return false;
508 }
