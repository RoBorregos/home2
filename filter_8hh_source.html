123 static constexpr const uint32_t MORTON_Y_MASK = 0x92492492;
124 static constexpr const uint32_t MORTON_Z_MASK = 0x24924924;
125 static constexpr const uint32_t EBMASK = 0x000000FF;

127 constexpr const uint32_t MORTON_FACTOR = 1000;

129 inline constexpr auto remap_point(float x, float min, float max) -> uint32_t
130 {
131 return ((x - min) / (max - min)) * static_cast<float>(MORTON_FACTOR);
132 }
133
134 inline auto morton_lut(uint32_t x, uint32_t y, uint32_t z) -> uint32_t
135 {
136 uint32_t answer = 0;
137 for (auto i = 4u; i > 0; --i)
138 {
139 const auto shift = (i - 1) * 8;
140 answer = answer << 24 |
141 (MORTON_LUT_Z_256[(z >> shift) & EBMASK] | MORTON_LUT_Y_256[(y >> shift) & EBMASK] |
142 MORTON_LUT_X_256[(x >> shift) & EBMASK]);
143 }
144
145 return answer;
146 }

159 // Filter a pointcloud into a smaller representative pointcloud.
160 //
161 // Inputs
162 // - `pc`: the initial pointcloud
163 // - `min_dist`: The minimum distance between two points to be considered distinct.
164 // - `max_range`: The maximum distance for a point from the origin to be retained from `pc`.
165 // - `origin`: The location of the origin.
166 // - `workspace_min`: The minimum vertex of the AABB describing the workspace.
167 // - `workspace_max`: The maximum vertex of the AABB describing the workspace.
168 //
169 // The AABB bounded by `workspace_min` and `workspace_max` must contain all points in `pc`.
170 //
171 // Returns a subset of `pc`, subject to the following conditions:
172 // - If there exists `p1`, `p2` in `pc`, such that `d(p1, p2) <= min_dist`, then one of `p1` or `p2` _may_
173 // be removed.
174 // - If `d(p, origin) >= max_range`, `p` will be removed.
175 template <typename PointCloud>
176 auto filter_pointcloud(
177 const PointCloud &pc,
178 float min_dist,
179 float max_range,
180 Point origin,
181 Point workspace_min,
182 Point workspace_max,
183 bool cull) -> std::vector<Point>
184 {
185 if (pc.shape(0) == 0)
186 {
187 return std::vector<Point>();
188 }
189
190 const auto sqdist = min_dist * min_dist;
191 const auto sqrange = max_range * max_range;
192 auto min = std::min({origin[0] - max_range, origin[1] - max_range, origin[2] - max_range});
193 auto max = std::min({origin[0] + max_range, origin[1] + max_range, origin[2] + max_range});
194
195 std::vector<std::pair<uint32_t, uint32_t>> morton;
196 morton.resize(pc.shape(0));
197
198 auto hi = 0U;
199 // Step 1: filter out any points so far from the origin that we wouldn't collide with them anyway.
200 for (auto i = 0U; i < pc.shape(0); ++i)
201 {
202 if (cull)
203 {
204 if (sql2_3(pc(i, 0), pc(i, 1), pc(i, 2), origin[0], origin[1], origin[2]) < sqrange and
205 workspace_min[0] <= pc(i, 0) and pc(i, 0) <= workspace_max[0] and
206 workspace_min[1] <= pc(i, 1) and pc(i, 1) <= workspace_max[1] and
207 workspace_min[2] <= pc(i, 2) and pc(i, 2) <= workspace_max[2])
208 {
209 morton[hi++].first = i;
210 }
211 }
212 else
213 {
214 morton[hi++].first = i;
215 }
216 }
217
218 std::array<unsigned int, 3> coordinates{0, 1, 2};
219
220 // For each space-filling curve...
221 do
222 {
223 // Step 2: Sort points according to their position in the space-filling curve.
224 float new_min = max;
225 float new_max = min;
226 for (auto &h : morton)
227 {
228 auto c0 = remap_point(pc(h.first, coordinates[0]), min, max);
229 auto c1 = remap_point(pc(h.first, coordinates[1]), min, max);
230 auto c2 = remap_point(pc(h.first, coordinates[2]), min, max);
231
232 new_min = std::min({new_min, pc(h.first, 0), pc(h.first, 1), pc(h.first, 2)});
233 new_max = std::max({new_max, pc(h.first, 0), pc(h.first, 1), pc(h.first, 2)});
234
235 h.second = morton_encode(c0, c1, c2);
236 }
237
238 pdqsort_branchless(
239 morton.begin(),
240 morton.end(),
241 [](const auto &a, const auto &b) { return a.second < b.second; });
242
243 std::vector<std::pair<uint32_t, uint32_t>> filtered;
244 filtered.reserve(morton.size());
245 filtered.emplace_back(morton.front());
246
247 // Step 3: Any adjacent pair of points in the sort which are too close get eliminated.
248 for (auto i = 1u; i < morton.size(); ++i)
249 {
250 const auto h1 = morton[i].first;
251 const auto h2 = filtered.back().first;
252
253 if (sql2_3(pc(h1, 0), pc(h1, 1), pc(h1, 2), pc(h2, 0), pc(h2, 1), pc(h2, 2)) > sqdist)
254 {
255 filtered.emplace_back(morton[i]);
256 }
257 }
258
259 morton = std::move(filtered);
260
261 max = (new_max + max) / 2.;
262 min = (new_min + min) / 2.;
263
264 } while (std::next_permutation(coordinates.begin(), coordinates.end()));
265
266 std::vector<Point> filtered;
267 filtered.reserve(morton.size());
268
269 for (const auto &h : morton)
270 {
271 filtered.emplace_back(Point{pc(h.first, 0), pc(h.first, 1), pc(h.first, 2)});
272 }
273
274 return filtered;
275 }
