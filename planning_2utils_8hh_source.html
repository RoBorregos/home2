76 inline auto astar(

78 const Configuration &start,
79 const Configuration &goal,

81 {

85 for (auto i = 0U; i < graph.size(); ++i)
86 {
87 parents[i] = std::numeric_limits<unsigned int>::max();
88 expanded[i] = false;
89 }

91 // NOTE: Assumes that the start is the first element of the graph and the goal is the second
92 constexpr const unsigned int start_index = 0;
93 constexpr const unsigned int goal_index = 1;
94 parents[start_index] = start_index;
95 std::vector<utils::QueueNode> open_set;
96 open_set.emplace_back(utils::QueueNode{start_index, goal.distance(start)});
97 Configuration temp;
98 while (not open_set.empty())
99 {
100 const auto current = open_set.back();
101 open_set.pop_back();
102 if (current.index == goal_index)
103 {
104 break;
105 }
106
107 if (expanded[current.index])
108 {
109 continue;
110 }
111 expanded[current.index] = true;
112
113 const auto &current_node = graph[current.index];
114 const auto current_g = current_node.g;
115 for (const auto &[neighbor_index, neighbor_distance] : current_node.neighbors)
116 {
117 auto &neighbor_node = graph[neighbor_index];
118 if (current_g + neighbor_distance >= neighbor_node.g)
119 {
120 continue;
121 }
122
123 neighbor_node.g = current_g + neighbor_distance;
124 parents[neighbor_index] = current_node.index;
125 temp = state_index(neighbor_index);
126 open_set.emplace_back(
127 utils::QueueNode{neighbor_index, neighbor_node.g + goal.distance(temp)});
128 }
129
130 pdqsort(
131 open_set.begin(),
132 open_set.end(),
133 [](const auto &a, const auto &b) { return a.cost > b.cost; });
134 }
135
136 if (parents[goal_index] == std::numeric_limits<unsigned int>::max())
137 {
138 return nullptr;
139 }

141 return parents;
142 }
